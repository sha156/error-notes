def set_font(run, name='仿宋_GB2312', size_pt=14, bold=False, color=None):
    run.font.name = name
    run.font.size = Pt(size_pt)
    run.bold = bold
    # Set East Asian font
    r = run._r
    rPr = r.get_or_add_rPr()
    rFonts = OxmlElement('w:rFonts')
    rFonts.set(qn('w:ascii'), name)
    rFonts.set(qn('w:eastAsia'), name)
    rFonts.set(qn('w:hAnsi'), name)
    rFonts.set(qn('w:cs'), '宋体')
    rPr.insert(0, rFonts)
    if color:
        run.font.color.rgb = RGBColor(*color)

def add_para(doc, text, align=WD_ALIGN_PARAGRAPH.CENTER, font_name='仿宋_GB2312', size_pt=14, bold=False, space_before=0, space_after=0, first_line_indent=0, color=None):
    p = doc.add_paragraph()
    p.alignment = align
    pf = p.paragraph_format
    pf.space_before = Pt(space_before)
    pf.space_after = Pt(space_after)
    if first_line_indent:
        pf.first_line_indent = Pt(first_line_indent)
    run = p.add_run(text)
    set_font(run, font_name, size_pt, bold, color)
    return p
